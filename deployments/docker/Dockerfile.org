# Build stage
FROM golang:1.24-alpine AS builder

WORKDIR /app

# Copy go mod files
COPY go.mod go.sum ./
RUN go mod download

# Copy source code
COPY . .

# Build the organization service
RUN CGO_ENABLED=0 GOOS=linux go build -o /bin/org-service ./services/org

# Runtime stage
FROM alpine:latest

RUN apk --no-cache add ca-certificates

WORKDIR /root/

# Copy the binary from builder
COPY --from=builder /bin/org-service .

EXPOSE 50054

CMD ["./org-service"]
